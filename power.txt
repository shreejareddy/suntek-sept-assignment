#include<stdio.h>
#include<stdlib.h>
 int power(int,int);
void main()
{
    int x,n,res=1,i,k;
    printf("enter number x");
    scanf("%d",&x);
    printf("enter power to be raised");
    scanf("%d",&n);
    //iterative
    for(i=1;i<=n;i++)
    {
        res=res*x;
    }
    printf("%d",res);
    k=power(x,n);
    printf("%d ",k);
}
//recursion
int power(int x,int n)
{
    if(n==0)
    return 1;
    else if(n>0)
    return x*power(x,n-1);
    else 
    exit(0);
}
